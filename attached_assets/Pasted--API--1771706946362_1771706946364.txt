### **فایل جدید: `test-all-apis.ts`**

```typescript
import axios from 'axios';

interface TestResult {
  service: string;
  key: string;
  status: 'موفق ✅' | 'خطا ❌';
  latency?: number;
  error?: string;
  model?: string;
}

const keys = {
  poyo: [
    'sk-G8djO1CepO_vfl0u5CDGDdD6dXC5zG67rX07RDUZadqQQ5zI627VTifWq5CsJm',
    'sk-NdIelDiC8dgJXP-uSy-4_03BQnGaCX1xdtVYZXFa9Z1b4FqXF3oProuUg9huz_'
  ],
  openrouter: [
    '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',
    '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'
  ],
  portkey: [
    'gASN7iokVzgqJLweJTWr12V75JG+'
  ]
};

// ================== تست PoYo AI ==================
async function testPoYo(apiKey: string, index: number): Promise<TestResult> {
  const start = Date.now();
  
  try {
    const response = await axios.post(
      'https://api.poyo.ai/v1/chat/completions',
      {
        model: 'gpt-4o-mini',
        messages: [{ role: 'user', content: 'سلام' }],
        max_tokens: 10
      },
      {
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json'
        },
        timeout: 10000
      }
    );

    return {
      service: 'PoYo AI',
      key: `کلید #${index + 1}`,
      status: 'موفق ✅',
      latency: Date.now() - start,
      model: 'gpt-4o-mini'
    };
  } catch (error: any) {
    return {
      service: 'PoYo AI',
      key: `کلید #${index + 1}`,
      status: 'خطا ❌',
      error: error.response?.data?.error?.message || error.message
    };
  }
}

// ================== تست OpenRouter ==================
async function testOpenRouter(apiKey: string, index: number): Promise<TestResult> {
  const start = Date.now();
  
  try {
    const response = await axios.post(
      'https://openrouter.ai/api/v1/chat/completions',
      {
        model: 'google/gemini-2.0-flash-exp:free',
        messages: [{ role: 'user', content: 'سلام' }],
        max_tokens: 10
      },
      {
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
          'HTTP-Referer': 'https://arman-law.com',
          'X-Title': 'Arman Law Firm'
        },
        timeout: 10000
      }
    );

    return {
      service: 'OpenRouter',
      key: `کلید #${index + 1}`,
      status: 'موفق ✅',
      latency: Date.now() - start,
      model: 'gemini-2.0-flash'
    };
  } catch (error: any) {
    return {
      service: 'OpenRouter',
      key: `کلید #${index + 1}`,
      status: 'خطا ❌',
      error: error.response?.data?.error?.message || error.message
    };
  }
}

// ================== تست Portkey ==================
async function testPortkey(apiKey: string, index: number): Promise<TestResult> {
  const start = Date.now();
  
  try {
    const response = await axios.post(
      'https://api.portkey.ai/v1/chat/completions',
      {
        model: 'gpt-4o-mini',
        messages: [{ role: 'user', content: 'سلام' }],
        max_tokens: 10
      },
      {
        headers: {
          'x-portkey-api-key': apiKey,
          'x-portkey-provider': 'openai',
          'Content-Type': 'application/json'
        },
        timeout: 10000
      }
    );

    return {
      service: 'Portkey AI',
      key: `کلید #${index + 1}`,
      status: 'موفق ✅',
      latency: Date.now() - start,
      model: 'gpt-4o-mini (via OpenAI)'
    };
  } catch (error: any) {
    return {
      service: 'Portkey AI',
      key: `کلید #${index + 1}`,
      status: 'خطا ❌',
      error: error.response?.data?.error?.message || error.message
    };
  }
}

// ================== اجرای همه تست‌ها ==================
async function runAllTests() {
  console.log('🧪 شروع تست همه API ها...\n');

  const results: TestResult[] = [];

  // تست PoYo
  for (let i = 0; i < keys.poyo.length; i++) {
    const result = await testPoYo(keys.poyo[i], i);
    results.push(result);
  }

  // تست OpenRouter
  for (let i = 0; i < keys.openrouter.length; i++) {
    const result = await testOpenRouter(keys.openrouter[i], i);
    results.push(result);
  }

  // تست Portkey
  for (let i = 0; i < keys.portkey.length; i++) {
    const result = await testPortkey(keys.portkey[i], i);
    results.push(result);
  }

  // نمایش جدول
  console.log('\n📊 نتایج تست:\n');
  console.table(results);

  // خلاصه
  const successful = results.filter(r => r.status === 'موفق ✅').length;
  const failed = results.filter(r => r.status === 'خطا ❌').length;

  console.log(`\n📈 خلاصه: ${successful} موفق، ${failed} ناموفق\n`);

  // نمایش خطاها
  const errors = results.filter(r => r.error);
  if (errors.length > 0) {
    console.log('🔴 خطاها:\n');
    errors.forEach(e => {
      console.log(`${e.service} ${e.key}: ${e.error}`);
    });
  }

  // بهترین گزینه
  const working = results.filter(r => r.status === 'موفق ✅');
  if (working.length > 0) {
    const fastest = working.sort((a, b) => (a.latency || 9999) - (b.latency || 9999))[0];
    console.log(`\n⚡ سریع‌ترین: ${fastest.service} ${fastest.key} (${fastest.latency}ms)\n`);
  }
}

// اجرا
runAllTests().catch(console.error);
```

### **روش 2: در داشبورد Next.js/React**

فایل `pages/api/test-apis.ts` بسازید:

```typescript
import type { NextApiRequest, NextApiResponse } from 'next';
import axios from 'axios';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  const results = [];

  // تست PoYo
  try {
    await axios.post('https://api.poyo.ai/v1/chat/completions', {
      model: 'gpt-4o-mini',
      messages: [{ role: 'user', content: 'test' }]
    }, {
      headers: { 'Authorization': `Bearer ${process.env.POYO_API_KEY}` }
    });
    results.push({ service: 'PoYo', status: '✅' });
  } catch (e: any) {
    results.push({ service: 'PoYo', status: '❌', error: e.message });
  }

  // تست OpenRouter
  try {
    await axios.post('https://openrouter.ai/api/v1/chat/completions', {
      model: 'google/gemini-2.0-flash-exp:free',
      messages: [{ role: 'user', content: 'test' }]
    }, {
      headers: { 'Authorization': `Bearer ${process.env.OPENROUTER_API_KEY}` }
    });
    results.push({ service: 'OpenRouter', status: '✅' });
  } catch (e: any) {
    results.push({ service: 'OpenRouter', status: '❌', error: e.message });
  }

  // تست Portkey
  try {
    await axios.post('https://api.portkey.ai/v1/chat/completions', {
      model: 'gpt-4o-mini',
      messages: [{ role: 'user', content: 'test' }]
    }, {
      headers: {
        'x-portkey-api-key': process.env.PORTKEY_API_KEY!,
        'x-portkey-provider': 'openai'
      }
    });
    results.push({ service: 'Portkey', status: '✅' });
  } catch (e: any) {
    results.push({ service: 'Portkey', status: '❌', error: e.message });
  }

  res.json(results);
}
```

بعد در فرانت‌اند:
